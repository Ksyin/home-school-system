rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() { return request.auth != null; }
    function myUserDoc() { return get(/databases/$(database)/documents/users/$(request.auth.uid)); }
    function myRole() { return signedIn() ? myUserDoc().data.role : null; }
    match /users/{userId} { allow create: if signedIn() && request.auth.uid == userId; allow read, update: if signedIn() && request.auth.uid == userId; }
    match /students/{studentId} { allow read: if signedIn(); allow create, update: if signedIn() && (myRole() == 'parent' || myRole() == 'tutor'); }
    match /assignments/{id} { allow read: if signedIn(); allow create, update: if signedIn() && myRole() == 'tutor'; }
    match /submissions/{id} { allow read: if signedIn(); allow create, update: if signedIn() && myRole() == 'student'; }
    match /assessments/{id}, /attendance/{id}, /reports/{id}, /lessonPlans/{id}, /classrooms/{id}, /resources/{id} { allow read: if signedIn(); allow create, update: if signedIn() && myRole() == 'tutor'; }
    match /portfolios/{id} { allow read: if signedIn(); allow create, update: if signedIn() && (myRole() == 'student' || myRole() == 'tutor'); }
    match /messages/{id} { allow read, create: if signedIn(); allow update: if false; }
  }
}
